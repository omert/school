\title{Coding Theory - Exercise 3 - Addendum - Linear Independence of 
Characters}

 \author{Omer Tamuz, 035696574}
\maketitle

\begin{theorem}
  Let $G$ be an Abelian group. Let $\chi_1,\ldots,\chi_n$ be distinct 
characters of $G$. Then $\chi_1,\ldots,\chi_n$ are linearly independent, so 
that 
\begin{equation*}
  \sum_{i=0}^{n}c_i\chi_i=0
\end{equation*}
implies that $c_i=0$ for all $i$.
\end{theorem}
\begin{proof}
We'll prove by induction. For $n=1$ the theorem is self-evident. Assume it
is true for $n-1$. Let
\begin{equation}
  \label{eq:sum_g}
  \sum_{i=0}^nc_i\chi_i(g)=0
\end{equation}
for some $c_i$'s and all $g\in G$. 
Assume $\chi_1(g_0)\neq \chi_n(g_0)$. Since the l.h.s of
Eq.~\ref{eq:sum_g} is zero for all $g$, then it is also zero when we
evaluate it at $g_0g$:
\begin{equation}
  \label{eq:sum_g0}
  \sum_{i=0}^nc_i\chi_i(g_0)\chi_i(g)=0  
\end{equation}
We now multiply Eq.~\ref{eq:sum_g} by $\chi_n(g_0)$ and subtract it from 
Eq.~\ref{eq:sum_g0}:
\begin{equation}
  \label{eq:sum_g0_g}
  \sum_{i=0}^{n-1}c_i(\chi_i(g_0)-\chi_n(g_0))\chi_i(g)=0.
\end{equation}
Note that the last term cancels, and therefore this is a linear independence
on $\chi_1$ to $\chi_{n-1}$. Hence, by the inductive assumption, all the 
coefficients are zero. Since $\chi_1(g_0)\neq\chi_n(g_0)$, we can conclude
that $c_1=0$. By similar argument (exchanging $c_i$ for $c_1$ throughout) we
can conclude that $c_i=0$ for $i=1,\ldots,n-1$. Hence Eq.~\ref{eq:sum_g}
becomes
\begin{equation}
  \label{eq:sum_g_n}
  c_n\chi_n(g)=0
\end{equation}
and since this must be true for all $g$, we can also conclude that $c_n=0$.
\end{proof}
